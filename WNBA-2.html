import matplotlib.pyplot as plt
import pandas as pd

# WNBA Finals viewership data
data = {
    'Year': [2020, 2021, 2022, 2023, 2024],
    'Viewership': [351000, 469000, 555000, 1300000, 3300000],
    'Matchup': ['Aces vs. Storm', 'Sky vs. Mercury', 'Sun vs. Aces', 'Liberty vs. Aces', 'Liberty vs. Lynx']
}

df = pd.DataFrame(data)

# Assign colors to each matchup
colors = {'Aces vs. Storm': 'red',
          'Sky vs. Mercury': 'blue',
          'Sun vs. Aces': 'green',
          'Liberty vs. Aces': 'orange',
          'Liberty vs. Lynx': 'purple'}

# Create the list visual
fig, ax = plt.subplots(figsize=(10, 8))

# Remove axes
ax.axis('off')

# Add list items with color coding and complex formatting
y_pos = 0.8  # Starting y-position
for i in range(len(df)):
    text = f"{df['Year'][i]}:\n"  # Add newline for year
    text += f"{' ' * 4}{df['Matchup'][i]}\n"  # Indent matchup
    text += f"{' ' * 8}Viewership: {df['Viewership'][i]:,}"  # Indent viewership
    ax.text(0.1, y_pos, text, color=colors[df['Matchup'][i]], fontsize=12,
            bbox=dict(facecolor='lightgray', edgecolor='black', boxstyle='round,pad=0.5'))  # Add box
    y_pos -= 0.2  # Adjust y-position for next item with more spacing

# Title
ax.set_title('WNBA Finals Viewership Increase', fontsize=14, pad=20)

plt.show()

import matplotlib.pyplot as plt
import pandas as pd

# WNBA Finals viewership data
data = {
    'Year': [2020, 2021, 2022, 2023, 2024],
    'Viewership': [351000, 469000, 555000, 1300000, 3300000],
    'Matchup': ['Aces vs. Storm', 'Sky vs. Mercury', 'Sun vs. Aces', 'Liberty vs. Aces', 'Liberty vs. Lynx']
}

df = pd.DataFrame(data)

# Assign colors to each matchup
colors = {'Aces vs. Storm': 'red',
          'Sky vs. Mercury': 'blue',
          'Sun vs. Aces': 'green',
          'Liberty vs. Aces': 'orange',
          'Liberty vs. Lynx': 'purple'}

# Create the bubble chart
plt.figure(figsize=(10, 6))
for i in range(len(df)):
    plt.scatter(df['Year'][i], df['Viewership'][i],
                s=df['Viewership'][i] / 10000,
                color=colors[df['Matchup'][i]],
                alpha=0.7,
                label=df['Matchup'][i])

    # Add viewership as text inside the bubble
    plt.text(df['Year'][i], df['Viewership'][i],
             f"{df['Viewership'][i]:,}",  # Format with commas for thousands separator
             ha='center', va='center', color='white', fontsize=8)  # Adjust fontsize as needed

plt.xlabel('Year')
plt.ylabel('Viewership')
plt.title('WNBA Finals Viewership Increase')
plt.xticks(df['Year'])
plt.legend(loc='upper left', title='Matchup')

plt.tight_layout()
plt.grid(True, linestyle='--', alpha=0.5)
plt.show()

Above we use a bubble chart to convey the wnba FINALS championship game increase of veiwership over the last five years.
